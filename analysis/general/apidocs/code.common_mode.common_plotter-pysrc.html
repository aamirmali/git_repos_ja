 55          #Returns (common,n_invalid), where common is the common mode of 
 56          #set_of_data, and n_invalid is the number of zero elements. If all 
 57          #elements are zero, returns (None,None) instead. The common mode is 
 58          #defined as the sum of all non-zero elements in set_of_data, either  
 59          #normalized or not depending on self.normalize, divided by the number 
 60          #of summed elements. 
 61          common=np.zeros(len(set_of_data[0])) 
 62          n_invalid=0 
 63          for element in set_of_data: 
 64              element -= np.mean(element) 
 65              norm=np.sqrt(np.dot(element,element)) 
 66              if norm==0: 
 67                  n_invalid += 1 
 68              else: 
 69                  if self.normalize: 
 70                      element /= norm 
 71                  common += element 
 72          if n_invalid==len(set_of_data):

 75          #normalized, units are meaningless anyways, so we might as well divide 
 76          #by the length of data
